Firmware architecturé pour ESP32. Gestion de capteurs ultrasoniques, environnementaux (DHT11) et actionneurs cinétiques via servo-moteur. Interface monitoring I2C et serveur Web asynchrone embarqué.

smart_bin_firmware.ino 
#include <WiFi.h>
#include <DHT.h>
#include <Adafruit_Sensor.h>
#include <ESP32Servo.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// WIFI
const char *ssid = "Abdoulaye";
const char *password = "00112233";

// DHT11
#define DHTPIN 4
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

// Ultrason
#define TRIG 5
#define ECHO 34

// LED
#define LED_VERTE 26
#define LED_ROUGE 27

// Buzzer
#define BUZZER 25

// Servo
Servo servo;
#define SERVO_PIN 13

// LCD
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Web
WiFiServer server(80);

float distance;

unsigned long lcdTimer = 0;
unsigned long porteTimer = 0;
unsigned long buzzerTimer = 0;

bool porteOuverte = false;
bool buzzerOn = false;

void setup()
{
  Serial.begin(115200);

  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);

  pinMode(LED_VERTE, OUTPUT);
  pinMode(LED_ROUGE, OUTPUT);

  pinMode(BUZZER, OUTPUT);

  servo.attach(SERVO_PIN);

  Wire.begin(32, 14);
  lcd.init();
  lcd.backlight();

  dht.begin();

  // WIFI
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED)
  {
    delay(1000);
    Serial.print(".");
  }

  Serial.println("WiFi Connecte");
  Serial.println(WiFi.localIP());

  server.begin();
}

void loop()
{
  // ===== DISTANCE =====

  digitalWrite(TRIG, LOW);
  delayMicroseconds(2);

  digitalWrite(TRIG, HIGH);
  delayMicroseconds(10);

  digitalWrite(TRIG, LOW);

  distance = pulseIn(ECHO, HIGH) * 0.034 / 2;

  // ===== DHT =====

  float h = dht.readHumidity();
  float t = dht.readTemperature();

  // ===== OUVERTURE PORTE =====

  if (distance <= 5 && !porteOuverte)
  {
    servo.write(90);

    digitalWrite(LED_VERTE, HIGH);
    digitalWrite(LED_ROUGE, LOW);

    porteTimer = millis();
    porteOuverte = true;

    digitalWrite(BUZZER, HIGH);
    buzzerTimer = millis();
    buzzerOn = true;
  }

  // ===== FERMETURE APRES 4s =====

  if (porteOuverte && millis() - porteTimer > 4000)
  {
    servo.write(0);

    digitalWrite(LED_VERTE, LOW);
    digitalWrite(LED_ROUGE, HIGH);

    porteOuverte = false;
  }

  // ===== ARRET BUZZER =====

  if (buzzerOn && millis() - buzzerTimer > 1000)
  {
    digitalWrite(BUZZER, LOW);
    buzzerOn = false;
  }

  // ===== LCD ALIGNÉ 16x2 =====

  if (millis() - lcdTimer > 2000)
  {
    lcd.clear();

    // Ligne 1
    lcd.setCursor(0, 0);

    if (porteOuverte)
      lcd.print("P:Ouverte ");
    else
      lcd.print("P:Fermee  ");

    lcd.print("T:");
    lcd.print((int)t);
    lcd.print("C");

    // Ligne 2
    lcd.setCursor(0, 1);

    lcd.print("H:");
    lcd.print((int)h);

    lcd.print(" D:");
    lcd.print((int)distance);
    lcd.print("cm");

    lcdTimer = millis();
  }

  // ===== WEB SERVER =====

  WiFiClient client = server.available();

  if (client)
  {
    String etat;

    if (porteOuverte)
      etat = "Ouverte";
    else
      etat = "Fermee";

    String currentLine = "";

    while (client.connected())
    {
      if (client.available())
      {
        char c = client.read();
        if (c == '\n')
        {
          if (currentLine.length() == 0)
          {
            String response = "HTTP/1.1 200 OK\r\n";
            response += "Content-Type:text/html\r\n";
            response += "Connection:close\r\n\r\n";

            response += "<!DOCTYPE HTML><html>";
            // ... [Server Response HTML] ...
            client.println(response);
            break;
          }
          else
            currentLine = "";
        }
        else if (c != '\r')
          currentLine += c;
      }
    }
    client.stop();
  }
  delay(200);
}
